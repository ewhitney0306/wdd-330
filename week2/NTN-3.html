Array Properties and Methods

const avengers = ['Captain America', 'Iron Man', 'Thor', 'Hulk', 'Hawkeye', 'Black Widow'];
avengers.length;
<< 6
avengers[avengers.length -1];
<<'Black Widow'
//length can be changed

POP, PUSH, SHIFT, UNSHIFT

// Pop removes the last item in the array
avengers.pop();
<<'Black Widow'
avengers
<<['Captain America', 'Iron Man', 'Thor', 'Hulk', 'Hawkeye'];
//Shift removes to first item in the array
avengers.shift();
<<'Captain America'
avengers
<<[Iron Man', 'Thor', 'Hulk', 'Hawkeye'];
//Push adds a value to the end of the array.
avengers.push('Captain America');
avengers
<<['Iron Man', 'Thor', 'Hulk', 'Hawkeye','Captain America'];
//Unshift adds a value to the beginning of the array
avengers.unshift('Black Widow');
avengers
<<['Black Widow', 'Iron Man', 'Thor', 'Hulk', 'Hawkeye','Captain America'];

Merging Arrays

//Concat creates a new array that combines one or more arrays
avengers.concat(['Spiderman', 'Vision', 'Wanda']);
<<['Captain America', 'Iron Man', 'Thor', 'Hulk', 'Hawkeye', 'Spiderman', 'Vision', 'Wanda']
//To use concat but not create a new array use assignment (=)
avengers = avengers.concat(['Spiderman', 'Vision', 'Wanda']);
avengers
<<['Captain America', 'Iron Man', 'Thor', 'Hulk', 'Hawkeye', 'Spiderman', 'Vision', 'Wanda']

The Join Method

// turns array into a string that comprises of all the items in the array
avengers.join();
<<'Captain America, Iron Man, Thor, Hulk, Hawkeye, Spiderman, Vision, Wanda'
avengers.join('&');
<<'Captain America & Iron Man & Thor & Hulk & Hawkeye & Spiderman & Vision & Wanda'

Slicing and Splicing

avengers.slice(2, 4);
<<['Thor', 'Hawkeye']
//This is nondistructive
avengers;
<<['Captain America', 'Iron Man', 'Thor', 'Hulk', 'Hawkeye', 'Spiderman', 'Vision', 'Wanda']
//splice is distructive. It replaces a value with something else
//First number is the index, the second number is how many to replace, and the third item is what to replace it with.
avengers.splice(3, 1, 'Dr. Strange');
avengers;
<<['Captain America', 'Iron Man', 'Thor', 'Dr. Strange', 'Hawkeye', 'Spiderman', 'Vision', 'Wanda']
avengers.splice(4, 0, 'Quicksilver');
avengers;
<<['Captain America', 'Iron Man', 'Thor', 'Dr. Strange', 'Quicksilver', 'Hawkeye', 'Spiderman', 'Vision', 'Wanda']

Reverse and Sort

avenger.reverse();
<<['Wanda', 'Vision', 'Spiderman', 'Hawkeye', 'Quicksilver', 'Dr. Strange', 'Thor', 'Iron Man', 'Captain America']
//Reverse and Sort change the array permanently
avenger.sort();
<<['Captain America', 'Dr. Strange', 'Hawkeye','Iron Man', 'Quicksilver', 'Spiderman', 'Thor', 'Vision', 'Vision','Wanda']
